Method Dispatch in Protocol

protocol P {
    func foo()
}

extension P {
    func foo() { ... }
    func bar() { ... }
}

struct A: P {
    func foo() { ... }
    func bar() { ... }
}

let a = A()
a.foo()   // A.foo
a.bar()   // A.bar

let p: P = A()
p.foo()   // A.foo
p.bar()   // P.bar
